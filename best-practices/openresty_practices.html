1 安装1

# requires
sudo apt-get install libreadline-dev libncurses5-dev libpcre3-dev libssl-dev perl make build-essential

tar xvf ngx_openresty-VERSION.tar.gz
cd ngx_openresty-VERSION/
# 单机开发启用 debug 方便调试
./configure --with-debug
make
sudo make install
